package formacaofcamaraex2;

import java.util.ArrayList;
import java.util.List;
import java.util.Scanner;

/**
 *
 * @author cipri
 */
public class Exercicio2 {

    /**
     * @param args the command line arguments
     */
    
    public static boolean primeNumber(int number){
        int count = 0;
        for (int i=2;i<=number;i++){
            if(number%i==0){
                count++;
                if (count > 1){
                    return false;
                }
            }
        }
        return true;
    }
    public static void main(String[] args) {
        // TODO code application logic here
        Scanner input = new Scanner (System.in);
        List results = new ArrayList();
        System.out.println("Digite o valor Inicial: ");
        int initialValue = input.nextInt();
        System.out.println("Digite o valor Final: ");
        int finalValue = input.nextInt();
        
        for (int i = initialValue; i <= finalValue; i++){
            if (primeNumber(i)){
                results.add(i);
            }
        }
        if (results.isEmpty()){
            System.out.println("Não há números primos nesse intervalo");
        }else{
            System.out.println("Números primos no intervalo: ");
            while (!results.isEmpty()){
                System.out.print(results.remove(0)+" ");
            }
        }
    }    
}
